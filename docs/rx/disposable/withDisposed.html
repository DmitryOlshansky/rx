    Example

    import rx;

auto sub = new SubjectObject!int;
size_t putCount = 0;
size_t disposedCount = 0;

auto disposable = sub.doSubscribe!(_ => putCount++)
    .withDisposed!(() => disposedCount++);

sub.put(1);
disposable.dispose();

assert(putCount == 1);
assert(disposedCount == 1);
